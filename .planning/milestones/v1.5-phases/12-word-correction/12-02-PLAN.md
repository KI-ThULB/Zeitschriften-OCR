    **New JS functions** (add after cancelEdit, before escapeHtml):

    ```js
    function editWord(word, span) {
      // Hide SVG overlay for this word while editing
      clearHighlight();

      editingSpan = span;

      // Measure span's current rendered width for input sizing
      const spanWidth = Math.max(60, span.offsetWidth);

      const input = document.createElement('input');
      input.type = 'text';
      input.className = 'word-input';
      input.value = word.content;
      input.style.width = spanWidth + 'px';
      input.dataset.wordId = word.id;

      const errorSpan = document.createElement('span');
      errorSpan.className = 'word-error';
      errorSpan.style.display = 'none';

      // Replace span content with input (keep span in DOM for cancel restore)
      span.textContent = '';
      span.appendChild(input);
      span.appendChild(errorSpan);

      input.focus();
      input.select();

      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          saveWord(word, span, input, errorSpan);
        }
        if (e.key === 'Escape') {
          e.preventDefault();
          cancelEdit();
        }
      });

      // Blur = cancel (silently restore)
      input.addEventListener('blur', () => {
        // Use setTimeout to let click on another word fire first,
        // so cancelEdit() from the onclick handler runs before blur
        setTimeout(() => {
          if (editingSpan === span) cancelEdit();
        }, 0);
      });
    }

    function cancelEdit() {
      if (!editingSpan) return;
      const span = editingSpan;
      const word = wordById[span.dataset.id];
      editingSpan = null;
      // Restore original text
      span.textContent = '';
      span.appendChild(document.createTextNode(word ? word.content : ''));
      // Restore SVG overlay for the word whose edit was cancelled
      showHighlight();
    }

    async function saveWord(word, span, input, errorSpan) {
      const raw = input.value;
      if (!raw.trim()) {
        errorSpan.textContent = 'Word cannot be empty';
        errorSpan.style.display = '';
        return;
      }
      errorSpan.style.display = 'none';

      const stem = stems[currentIndex];
      let resp;
      try {
        resp = await fetch(`/save/${encodeURIComponent(stem)}`, {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({word_id: word.id, content: raw}),
        });
      } catch (err) {
        errorSpan.textContent = 'Save failed — network error';
        errorSpan.style.display = '';
        return;
      }

      if (!resp.ok) {
        errorSpan.textContent = 'Save failed — invalid content';
        errorSpan.style.display = '';
        return;
      }

      // Success: update in-memory word, restore span with new text, flash green
      word.content = raw;
      editingSpan = null;
      span.textContent = '';
      span.appendChild(document.createTextNode(raw));
      // Trigger green flash — remove class first to allow re-animation if saved twice
      span.classList.remove('word-saved');
      void span.offsetWidth; // force reflow
      span.classList.add('word-saved');
      span.addEventListener('animationend', () => span.classList.remove('word-saved'), {once: true});

      // Restore SVG overlay on the now-corrected word span
      showHighlight();

      // Return focus to text panel
      document.getElementById('text-panel').focus();
    }
    ```
