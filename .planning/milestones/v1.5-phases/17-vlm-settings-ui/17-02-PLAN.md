**JS** — add to the `<script>` block (after `downloadMets()`, before the closing `</script>`):

```js
// ---------------------------------------------------------------------------
// VLM Settings
// ---------------------------------------------------------------------------

const BACKEND_PRESETS = {
  openwebui: {
    baseUrl: 'https://openwebui-workshop.test.uni-jena.de/ollama/v1',
    models: [
      'llama3.2-vision:latest',
      'bakllava:latest',
      'minicpm-v:latest',
      'moondream:latest',
      'granite3.2-vision:latest',
    ],
  },
  openrouter: {
    baseUrl: 'https://openrouter.ai/api/v1',
    models: [
      'google/gemini-flash-1.5',
      'openai/gpt-4o',
      'anthropic/claude-3-5-sonnet',
      'meta-llama/llama-3.2-90b-vision-instruct',
      'google/gemini-pro-1.5',
      'anthropic/claude-3-opus',
      'openai/gpt-4o-mini',
      'mistralai/pixtral-12b',
      'qwen/qwen-vl-max',
      'anthropic/claude-3-haiku',
    ],
  },
};

function getSelectedBackend() {
  const radio = document.querySelector('input[name="vlm-backend"]:checked');
  return radio ? radio.value : 'openwebui';
}

function populateModelDropdown(models, selectedModel) {
  const select = document.getElementById('vlm-model');
  select.innerHTML = '';
  for (const m of models) {
    const opt = document.createElement('option');
    opt.value = m;
    opt.textContent = m;
    if (m === selectedModel) opt.selected = true;
    select.appendChild(opt);
  }
}

function onBackendChange() {
  const backend = getSelectedBackend();
  const preset = BACKEND_PRESETS[backend];
  document.getElementById('vlm-base-url').value = preset.baseUrl;
  populateModelDropdown(preset.models, preset.models[0]);
  document.getElementById('settings-status').textContent = '';
}

async function loadModels() {
  const baseUrl = document.getElementById('vlm-base-url').value.trim();
  const apiKey = document.getElementById('vlm-api-key').value.trim();
  const status = document.getElementById('settings-status');
  status.className = '';
  status.textContent = 'Loading models\u2026';

  const url = '/settings/models?base_url=' + encodeURIComponent(baseUrl)
            + '&api_key=' + encodeURIComponent(apiKey);
  try {
    const resp = await fetch(url);
    const data = await resp.json();
    if (!resp.ok) {
      // Fall back to curated list
      const backend = getSelectedBackend();
      populateModelDropdown(BACKEND_PRESETS[backend].models, null);
      status.className = 'err';
      status.textContent = 'Could not load live models \u2014 showing curated list. ('
        + (data.error || resp.status) + ')';
      return;
    }
    const models = data.models || [];
    if (models.length === 0) {
      status.className = 'err';
      status.textContent = 'No models returned by endpoint.';
      return;
    }
    const currentModel = document.getElementById('vlm-model').value;
    populateModelDropdown(models, currentModel);
    status.className = 'ok';
    status.textContent = models.length + ' models loaded.';
  } catch (err) {
    status.className = 'err';
    status.textContent = 'Network error: ' + err.message;
  }
}

async function saveSettings() {
  const baseUrl = document.getElementById('vlm-base-url').value.trim();
  const apiKey = document.getElementById('vlm-api-key').value.trim();
  const model = document.getElementById('vlm-model').value;
  const status = document.getElementById('settings-status');
  status.className = '';
  status.textContent = 'Saving\u2026';

  try {
    const resp = await fetch('/settings', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        backend: 'openai_compatible',
        base_url: baseUrl,
        api_key: apiKey,
        model: model,
      }),
    });
    const data = await resp.json();
    if (!resp.ok) {
      status.className = 'err';
      status.textContent = 'Save failed: ' + (data.error || resp.status);
      return;
    }
    status.className = 'ok';
    status.textContent = '\u2713 Settings saved';
    setTimeout(() => {
      if (status.textContent === '\u2713 Settings saved') status.textContent = '';
    }, 3000);
  } catch (err) {
    status.className = 'err';
    status.textContent = 'Network error: ' + err.message;
  }
}

async function initSettings() {
  try {
    const resp = await fetch('/settings');
    if (!resp.ok) return;
    const data = await resp.json();
    if (!data.backend) return;  // empty settings

    // Restore base_url
    if (data.base_url) {
      document.getElementById('vlm-base-url').value = data.base_url;
      // Select matching backend radio if possible
      const isOpenRouter = data.base_url.includes('openrouter.ai');
      const radioVal = isOpenRouter ? 'openrouter' : 'openwebui';
      const radio = document.querySelector(`input[name="vlm-backend"][value="${radioVal}"]`);
      if (radio) radio.checked = true;
    }

    // Restore api_key (masked — just mark as having a value)
    if (data.api_key) {
      document.getElementById('vlm-api-key').value = data.api_key;
    }

    // Restore model — populate from preset, then set saved value
    const backend = getSelectedBackend();
    const preset = BACKEND_PRESETS[backend];
    const savedModel = data.model || '';
    const allModels = savedModel && !preset.models.includes(savedModel)
      ? [savedModel, ...preset.models]
      : preset.models;
    populateModelDropdown(allModels, savedModel || preset.models[0]);

  } catch (_) { /* silent — best-effort init */ }
}
```
