**JS** — add the `downloadMets()` function to the `<script>` block (near the end, before the closing `</script>`):

```js
async function downloadMets() {
  const toast = document.getElementById('mets-toast');
  toast.classList.remove('visible');
  toast.textContent = '';

  const resp = await fetch('/mets');

  if (resp.status === 204) {
    toast.textContent = 'No processed files yet — run OCR first.';
    toast.classList.add('visible');
    return;
  }

  if (!resp.ok) {
    toast.textContent = `Export failed (${resp.status}).`;
    toast.classList.add('visible');
    return;
  }

  // Trigger browser download
  const blob = await resp.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'mets.xml';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);
}
```
