**Step 1c: Add JavaScript functions**

Add the following functions to the `<script>` block, after the existing `loadSegments()` function
(before the closing `</script>` tag):

```javascript
// ── Article Browser ────────────────────────────────────────────────────────

let currentArticles = [];   // regions array for current stem
let activeArticleId = null; // region_id of highlighted article card

async function loadArticles(stem) {
  currentArticles = [];
  activeArticleId = null;
  renderArticleList([]);  // clear previous
  try {
    const resp = await fetch(`/articles/${encodeURIComponent(stem)}`);
    if (!resp.ok) return;  // 404 = not yet segmented — silently skip
    const data = await resp.json();
    currentArticles = data.regions || [];
    renderArticleList(currentArticles);
  } catch (e) {
    // Network error — silently skip (article list is optional)
  }
}

function renderArticleList(regions) {
  const panel = document.getElementById('text-panel');
  // Remove previous article section if present
  const existing = document.getElementById('article-section');
  if (existing) existing.remove();
  if (!regions || regions.length === 0) return;

  const section = document.createElement('div');
  section.id = 'article-section';

  const heading = document.createElement('h3');
  heading.textContent = `Articles (${regions.length})`;
  section.appendChild(heading);

  regions.forEach(region => {
    const card = document.createElement('div');
    card.className = 'article-card';
    card.dataset.regionId = region.id;

    const typeEl = document.createElement('div');
    typeEl.className = 'article-type';
    typeEl.textContent = region.type || 'unknown';

    const titleEl = document.createElement('div');
    titleEl.textContent = region.title || '(no title)';

    card.appendChild(typeEl);
    card.appendChild(titleEl);
    card.addEventListener('click', () => selectArticle(region, card));
    section.appendChild(card);
  });

  panel.appendChild(section);
}

function selectArticle(region, cardEl) {
  // Deactivate previously active card
  document.querySelectorAll('.article-card.active').forEach(c => c.classList.remove('active'));
  if (activeArticleId === region.id) {
    // Second click on same card — deselect
    activeArticleId = null;
    clearArticleHighlight();
    return;
  }
  activeArticleId = region.id;
  cardEl.classList.add('active');
  highlightRegion(region);
}

function highlightRegion(region) {
  if (!region || !region.bounding_box) return;
  const W = jpeg_width;
  const H = jpeg_height;
  if (!W || !H) return;

  const bb = region.bounding_box;
  const x = bb.x * W;
  const y = bb.y * H;
  const w = bb.width * W;
  const h = bb.height * H;

  const rect = document.getElementById('article-highlight-rect');
  rect.setAttribute('x', x);
  rect.setAttribute('y', y);
  rect.setAttribute('width', w);
  rect.setAttribute('height', h);
  rect.style.display = '';
}

function clearArticleHighlight() {
  document.getElementById('article-highlight-rect').style.display = 'none';
}

function highlightRegionById(regionId) {
  if (!regionId) return;
  const region = currentArticles.find(r => r.id === regionId);
  if (!region) return;
  const card = document.querySelector(`.article-card[data-region-id="${CSS.escape(regionId)}"]`);
  selectArticle(region, card || document.createElement('div'));
}
```

```html
    function escapeHtml(str) {
      return String(str)
        .replace(/&/g, '&amp;').replace(/</g, '&lt;')
        .replace(/>/g, '&gt;').replace(/"/g, '&quot;');
    }

    function doSearch() {
      const q = document.getElementById('search-input').value.trim();
      if (!q) return;
      const url = new URL(window.location.href);
      url.searchParams.set('q', q);
      window.location.href = url.toString();

    async function runSearch(q) {
      const countEl = document.getElementById('results-count');
      const listEl = document.getElementById('results-list');
      countEl.textContent = 'Searching…';
      listEl.innerHTML = '';

      try {
        const resp = await fetch('/search?q=' + encodeURIComponent(q));
        const data = await resp.json();
        const results = data.results || [];

        if (results.length === 0) {
          countEl.textContent = '';
          listEl.innerHTML = '<p id="no-results">No articles found for "' + escapeHtml(q) + '".</p>';
          return;
        }

        countEl.textContent = results.length + ' result' + (results.length !== 1 ? 's' : '') + ' for "' + escapeHtml(q) + '"';

        listEl.innerHTML = results.map(r => `
          <a class="result-item" href="/viewer/${encodeURIComponent(r.stem)}#${encodeURIComponent(r.region_id)}">
            <div class="result-stem">${escapeHtml(r.stem)}</div>
            <div class="result-type">${escapeHtml(r.type)}</div>
            <div class="result-title">${escapeHtml(r.title)}</div>
          </a>
        `).join('');
      } catch (err) {
        countEl.textContent = '';
        listEl.innerHTML = '<p id="no-results">Search failed — ' + escapeHtml(String(err)) + '</p>';
      }
    }

    document.addEventListener('DOMContentLoaded', () => {
      const params = new URLSearchParams(window.location.search);
      const q = params.get('q') || '';
      document.getElementById('search-input').value = q;
      if (q) runSearch(q);
    });
  </script>
</body>
</html>
```

**Step 2d: Add Flask route for search.html in app.py**

viewer routes (before `if __name__ == '__main__'`):


**Important:** This route conflicts with the `GET /search` JSON API endpoint added in Plan 18-01.
The resolution: the API endpoint added in 18-01 is named `search_articles` and returns JSON.
This new route returns HTML only when no `q` param is present OR when the `Accept` header prefers
HTML. However, the simplest approach that avoids conflict: **rename the API endpoint path** to
`/api/search` in both app.py and search.html's fetch call.

Alternatively, use content negotiation: if `request.args.get('q')` is present AND the request
comes from `fetch()` (check `Accept: application/json` or `X-Requested-With`), return JSON;
otherwise serve the HTML template.

**Recommended clean approach:** Keep `/search` as the HTML page route; move the JSON API to
`/api/search?q=`. Update search.html to fetch from `/api/search?q=...`. Update the route in
app.py: rename `search_articles()` to serve at `GET /api/search`.
