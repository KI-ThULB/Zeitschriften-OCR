**renderSidebar:**

```javascript
function renderSidebar(stems) {
  const sidebar = document.getElementById('sidebar');
  sidebar.innerHTML = stems.map((stem, i) =>
    `<div class="file-item" data-index="${i}">${escapeHtml(stem)}</div>`
  ).join('');
  sidebar.addEventListener('click', (e) => {
    const item = e.target.closest('.file-item');
    if (!item) return;
    loadFile(parseInt(item.dataset.index, 10));
  });
}
```

**loadFile — two-fetch sequence with generation counter (RESEARCH.md Pitfall 4):**

```javascript
async function loadFile(index) {
  if (index < 0 || index >= stems.length) return;
  currentIndex = index;
  const gen = ++loadGen;
  const stem = stems[index];

  // Update sidebar active state
  document.querySelectorAll('#sidebar .file-item').forEach((el, i) => {
    el.classList.toggle('active', i === index);
  });

  // Update Prev/Next button state
  document.getElementById('btn-prev').disabled = (index === 0);
  document.getElementById('btn-next').disabled = (index === stems.length - 1);

  // Clear previous state
  clearHighlight();
  activeWord = null;
  document.getElementById('text-panel').innerHTML = '<em>Loading…</em>';
  document.getElementById('image-panel').scrollTop = 0;
  document.getElementById('text-panel').scrollTop = 0;

  // Set image src — browser loads async; word clicks enabled after img.load event
  const img = document.getElementById('tiff-img');
  img.src = `/image/${stem}`;

  // Fetch ALTO data in parallel with image load
  try {
    const data = await fetch(`/alto/${stem}`).then(r => r.json());
    if (gen !== loadGen) return;  // superseded by a newer loadFile call
    pageWidth = data.page_width;
    pageHeight = data.page_height;
    wordById = Object.fromEntries(data.words.map(w => [w.id, w]));
    renderWords(data.words);
  } catch (err) {
    if (gen !== loadGen) return;
    document.getElementById('text-panel').innerHTML = `<em>Error loading ALTO: ${escapeHtml(String(err))}</em>`;
  }
}
```

**renderWords — batch DOM build via innerHTML (RESEARCH.md Pattern 6; event delegation):**

```javascript
function renderWords(words) {
  const panel = document.getElementById('text-panel');
  if (words.length === 0) {
    panel.innerHTML = '<em>No words found in this file.</em>';
    return;
  }
  panel.innerHTML = words.map(w =>
    `<span class="word" data-id="${escapeHtml(w.id)}">${escapeHtml(w.content)}</span> `
  ).join('');
  panel.onclick = (e) => {
    const span = e.target.closest('.word');
    if (!span) return;
    const word = wordById[span.dataset.id];
    if (!word) return;
    selectWord(word, span);
  };
}
```

**selectWord — draw SVG rect, scroll image panel, highlight span (locked decisions):**

Highlight appears on click only (not hover). Selected word: bold + yellow background. One word selected at a time. Clicking a word scrolls the image panel so the rect is visible. Bidirectional: clicking the rect scrolls to the span.

```javascript
function selectWord(word, span) {
  // Deselect previous
  document.querySelectorAll('.word.selected').forEach(s => s.classList.remove('selected'));
  span.classList.add('selected');
  activeWord = word;

  showHighlight(word);

  const rect = document.getElementById('highlight-rect');
  rect.dataset.wordId = word.id;
  rect.style.pointerEvents = 'all';

  // Scroll image panel so the highlight rect is visible
  rect.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}
```

**showHighlight / clearHighlight — SVG rect attribute update (RESEARCH.md Pattern 4):**

Scale factors use img.clientWidth / pageWidth (NOT naturalWidth). Only HPOS/VPOS are scaled by x/y factors respectively; WIDTH/HEIGHT likewise scaled.

```javascript
function showHighlight(word) {
  const img = document.getElementById('tiff-img');
  if (!img.clientWidth || !pageWidth || !pageHeight) return;
  const scaleX = img.clientWidth / pageWidth;
  const scaleY = img.clientHeight / pageHeight;
  const rect = document.getElementById('highlight-rect');
  rect.setAttribute('x', word.hpos * scaleX);
  rect.setAttribute('y', word.vpos * scaleY);
  rect.setAttribute('width', word.width * scaleX);
  rect.setAttribute('height', word.height * scaleY);
  rect.style.display = '';
}

function clearHighlight() {
  const rect = document.getElementById('highlight-rect');
  rect.style.display = 'none';
  rect.style.pointerEvents = 'none';
  activeWord = null;
}
```

**setupResizeObserver — reposition active rect on image resize (locked: RESEARCH.md Pattern 5):**

Attach once on DOMContentLoaded. Guard `img.clientWidth === 0` per RESEARCH.md Pitfall 2.

```javascript
function setupResizeObserver() {
  const img = document.getElementById('tiff-img');
  const observer = new ResizeObserver(() => {
    if (img.clientWidth === 0) return;
    if (activeWord !== null) showHighlight(activeWord);
  });
  observer.observe(img);
}
```

**Bidirectional: rect click scrolls to span (locked decision from CONTEXT.md):**

Wire the rect click listener once, after DOM is ready. The rect carries `data-word-id` set by selectWord.

```javascript
document.getElementById('highlight-rect').addEventListener('click', () => {
  const rect = document.getElementById('highlight-rect');
  const span = document.querySelector(`.word[data-id="${rect.dataset.wordId}"]`);
  if (!span) return;
  document.querySelectorAll('.word.selected').forEach(s => s.classList.remove('selected'));
  span.classList.add('selected');
  span.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
});
```

This listener is set up inside DOMContentLoaded (or inline at end of script). Note: the SVG overlay has `pointer-events:none` but the rect explicitly sets `pointer-events:all` when visible (via selectWord), so this click listener fires correctly per RESEARCH.md Pattern 7.

**setupKeyboard — Left/Right arrow navigation with INPUT guard (locked decision):**

```javascript
function setupKeyboard() {
  document.addEventListener('keydown', (e) => {
    if (document.activeElement.tagName === 'INPUT') return;
    if (e.key === 'ArrowLeft') navigateTo(currentIndex - 1);
    if (e.key === 'ArrowRight') navigateTo(currentIndex + 1);
  });
}
```

**escapeHtml — required for OCR content that may contain `<`, `>`, `&` (RESEARCH.md Don't Hand-Roll):**

```javascript
function escapeHtml(str) {
  return String(str)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}
```

**Implementation notes:**
- No frontend framework, no CDN, no build step — vanilla JS only per RESEARCH.md recommendation
- CSS uses flexbox (Claude's discretion — simpler than grid for this fixed three-column layout)
